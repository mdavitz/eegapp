const express = require('express'); const bodyParser = require('body-parser'); const { Configuration, OpenAIApi } = require("openai"); require('dotenv').config(); // For managing your API key securely const app = express(); app.use(bodyParser.json()); const configuration = new Configuration({ apiKey: process.env.OPENAI_API_KEY, // Store your API key securely }); const openai = new OpenAIApi(configuration); app.post('/api/chat', async (req, res) => { try { const userMessage = req.body.message; // Add a system prompt to guide the AI to answer EEG questions const prompt = `You are an expert in EEG and neurology. Answer the following question in a clear, concise, and professional manner. Question: ${userMessage}`; const response = await openai.createChatCompletion({ model: "gpt-4", // or your preferred model messages: [{ role: "system", content: prompt }], temperature: 0.7, }); res.json({ reply: response.data.choices[0].message.content }); } catch (error) { console.error("Error:", error); res.status(500).json({ reply: "There was an error processing your request." }); } }); app.listen(3000, () => { console.log("Server running on port 3000"); });
